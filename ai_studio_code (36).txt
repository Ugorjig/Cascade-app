import React from 'react';
import type { Product } from '../types.ts';
import { MOCK_PRODUCTS, CartIcon, BackIcon } from '../constants.tsx';

interface ShopPageProps {
  handleAddToCart: (product: Product) => void;
  isEmbedded?: boolean;
  openCartModal?: () => void;
  showCart?: boolean;
  products?: Product[];
  onBack?: () => void;
}

const ShopPage: React.FC<ShopPageProps> = ({ handleAddToCart, isEmbedded = false, openCartModal, showCart, products, onBack }) => {
  const productsToDisplay = products || MOCK_PRODUCTS;
  
  const shopContent = (
    <div className="p-4 grid grid-cols-2 gap-4">
      {productsToDisplay.length > 0 ? productsToDisplay.map(product => (
        <div key={product.id} className="bg-surface dark:bg-dark-surface rounded-lg overflow-hidden border border-gray-200 dark:border-dark-border">
          <img src={product.imageUrl} alt={product.name} className="w-full h-48 object-cover" />
          <div className="p-4">
            <h2 className="font-bold truncate">{product.name}</h2>
            <p className="text-sm text-on-surface-secondary dark:text-dark-on-surface-secondary">by {product.seller.name}</p>
            <div className="flex justify-between items-center mt-4">
              <p className="font-bold text-lg text-primary">${product.price.toFixed(2)}</p>
              <button
                onClick={() => handleAddToCart(product)}
                className="bg-primary text-white font-bold px-4 py-2 rounded-full hover:bg-primary/90 transition-colors duration-200 text-sm"
              >
                Add
              </button>
            </div>
          </div>
        </div>
      )) : (
        <div className="col-span-2 text-center p-8 text-on-surface-secondary dark:text-dark-on-surface-secondary">
            <p className="font-bold text-xl">The store is empty</p>
        </div>
      )}
    </div>
  );

  if (isEmbedded) {
    return shopContent;
  }
  
  return (
    <div className="w-full pb-16 md:pb-0">
      <div className="sticky top-0 bg-background/80 backdrop-blur-md z-10 px-4 py-3 border-b border-gray-200 flex justify-between items-center">
        <div className="flex items-center gap-4">
            {onBack && (
                <button onClick={onBack} className="p-2 -ml-2 rounded-full hover:bg-surface-hover dark:hover:bg-dark-surface-hover md:hidden">
                    <BackIcon className="w-6 h-6" />
                </button>
            )}
            <div>
              <h1 className="text-xl font-bold">Store</h1>
              <p className="text-sm text-on-surface-secondary">Discover products from creators</p>
            </div>
        </div>
        {showCart && (
          <button onClick={openCartModal} className="p-2 rounded-full hover:bg-gray-100" aria-label="Open cart">
            <CartIcon className="w-6 h-6 text-on-surface" />
          </button>
        )}
      </div>
      {shopContent}
    </div>
  );
};

export default ShopPage;
