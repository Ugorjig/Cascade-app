  post: PostType;
  currentUser: User;
  allUsers: User[];
  subscribedToUserIds: string[];
  handleToggleLike: (postId: string) => void;

const PollDisplay: React.FC<{ initialPoll: Poll }> = ({ initialPoll }) => {
    const [poll, setPoll] = useState(initialPoll);
    const [votedChoice, setVotedChoice] = useState<number | null>(null);
    const [isPollEnded, setIsPollEnded] = useState(false);
    
    useEffect(() => {
        const pollEndTime = new Date(poll.endsAt).getTime();
        const checkPollEnd = () => {
            if (new Date().getTime() > pollEndTime) {
                setIsPollEnded(true);
            }
        };
        checkPollEnd();
        const interval = setInterval(checkPollEnd, 60000); // Check every minute
        return () => clearInterval(interval);
    }, [poll.endsAt]);

    const handleVote = (choiceIndex: number) => {
        if (votedChoice !== null || isPollEnded) return;
        
        setVotedChoice(choiceIndex);
        const newChoices = poll.choices.map((choice, index) => {
            if (index === choiceIndex) {
                return { ...choice, votes: choice.votes + 1 };
            }
            return choice;
        });
        setPoll({ ...poll, choices: newChoices });
    };

    const totalVotes = poll.choices.reduce((sum, choice) => sum + choice.votes, 0);

    const getTimeRemaining = () => {
        const now = new Date().getTime();
        const endTime = new Date(poll.endsAt).getTime();
        const distance = endTime - now;
        if (distance < 0) return "Poll ended";
        
        const days = Math.floor(distance / (1000 * 60 * 60 * 24));
        const hours = Math.floor((distance % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
        const minutes = Math.floor((distance % (1000 * 60 * 60)) / (1000 * 60));
        
        if (days > 0) return `${days}d left`;
        if (hours > 0) return `${hours}h left`;
        return `${minutes}m left`;
    };

    const hasVoted = votedChoice !== null;

    return (
        <div className="mt-3 space-y-2">
            {poll.choices.map((choice, index) => {
                const percentage = totalVotes > 0 ? (choice.votes / totalVotes) * 100 : 0;
                const isVotedOption = votedChoice === index;

                return (
                    <div key={index}>
                        {hasVoted || isPollEnded ? (
                            <div className="relative w-full h-9 rounded-md border border-gray-300 dark:border-dark-border overflow-hidden">
                                <div className="absolute inset-0 bg-primary/20" style={{ width: `${percentage}%` }}></div>
                                <div className="absolute inset-0 px-3 flex items-center justify-between text-sm font-semibold">
                                    <span className="flex items-center gap-2">
                                        <span>{choice.text}</span>
                                        {isVotedOption && <span className="text-primary">✓</span>}
                                    </span>
                                    <span>{percentage.toFixed(0)}%</span>
                                </div>
                            </div>
                        ) : (
                            <button
                                onClick={() => handleVote(index)}
                                className="w-full text-left font-semibold px-4 py-2 border-2 border-gray-300 dark:border-dark-border rounded-md hover:border-primary dark:hover:border-primary transition-colors duration-200"
                            >
                                {choice.text}
                            </button>
                        )}
                    </div>
                );
            })}
            <div className="text-sm text-on-surface-secondary dark:text-dark-on-surface-secondary">
                <span>{totalVotes.toLocaleString()} votes</span>
                {!isPollEnded && <span> · {getTimeRemaining()}</span>}
            </div>
        </div>
    );
};


export const Post: React.FC<PostProps> = ({ post, currentUser, allUsers, subscribedToUserIds, handleToggleLike, handleToggleEcho, handleToggleBookmark, handleSubscribe, openGiftModal, handleAddToCart, viewCommunity, handleSearch, blockedUserIds, handleToggleBlock, handleViewPost, handleOpenQuoteModal, handleOpenEditModal, handleOpenBoostModal, handleViewProfile, handlePlayVideo, playingVideoId, handleShareToCommunity, isDetailView = false, isBoosted = false }) => {
  const [isAnalyticsModalOpen, setAnalyticsModalOpen] = useState(false);
  const [isMenuOpen, setMenuOpen] = useState(false);
  const menuRef = useRef<HTMLDivElement>(null);
  const [isShareMenuOpen, setShareMenuOpen] = useState(false);
  const { addNotification } = useNotifications();
  const { t } = useLanguage();
  const videoRef = useRef<HTMLVideoElement>(null);
  const isPlaying = post.id === playingVideoId;
  const isScheduled = post.status === 'scheduled' && post.scheduledAt && new Date(post.scheduledAt) > new Date();
  const [modalRoot, setModalRoot] = useState<HTMLElement | null>(null);

  const [isImageModalOpen, setImageModalOpen] = useState(false);
  const [isMuted, setIsMuted] = useState(true);
  const [currentTime, setCurrentTime] = useState(0);
  const [videoDuration, setVideoDuration] = useState(0);
  const [showControls, setShowControls] = useState(false);
  const controlsTimeoutRef = useRef<number | null>(null);
  const progressRef = useRef<HTMLDivElement>(null);

  const analyticsData = useMemo<PostAnalytics | null>(() => {
    if (post.analytics) {
      return post.analytics;
    }
    if (typeof post.views !== 'undefined' && post.views > 0) {
      return {
        impressions: post.views,
        engagements: (post.likes || 0) + (post.comments || 0) + (post.echos || 0),
        profileVisits: 0, // Not available on post object, default to 0
        newFollowers: 0, // Not available on post object, default to 0
      };
    }
    return null;
  }, [post]);

  useEffect(() => {
    // This effect runs on the client, so `document` is available.
    setModalRoot(document.getElementById('modal-root'));
  }, []);

  useEffect(() => {
    const videoElement = videoRef.current;
    if (videoElement) {
      if (isPlaying) {
        const playPromise = videoElement.play();
        if (playPromise !== undefined) {
          playPromise.catch(error => {
            if (error.name !== 'AbortError') {
              console.error("Video play failed:", error);
            }
          });
        }
      } else {
        videoElement.pause();
      }
    }
    return () => {
      if (videoElement) {
        videoElement.pause();
      }
    };
  }, [isPlaying]);

  const handleUserActivity = useCallback(() => {
    setShowControls(true);
    if (controlsTimeoutRef.current) {
      clearTimeout(controlsTimeoutRef.current);
    }
    controlsTimeoutRef.current = window.setTimeout(() => {
      if (isPlaying) { // Only hide if playing
        setShowControls(false);
      }
    }, 3000);
  }, [isPlaying]);

  useEffect(() => {
    if (!isPlaying) {
      setShowControls(true); // Keep controls visible when paused
      if (controlsTimeoutRef.current) {
        clearTimeout(controlsTimeoutRef.current);
      }
    } else {
        // When play starts, show controls then start timer to hide them
        handleUserActivity();
    }
    
    // Cleanup timeout on component unmount
    return () => {
      if (controlsTimeoutRef.current) {
        clearTimeout(controlsTimeoutRef.current);
      }
    }
  }, [isPlaying, handleUserActivity]);
  
  useEffect(() => {
    if (videoRef.current) {
      videoRef.current.muted = isMuted;
    }
  }, [isMuted]);

  // Video event handlers
  const handleTimeUpdate = () => {
    if (videoRef.current) {
      setCurrentTime(videoRef.current.currentTime);
    }
  };
  const handleLoadedMetadata = () => {
    if (videoRef.current) {
      setVideoDuration(videoRef.current.duration);
    }
  };
  const handleVideoEnd = () => {
      if (videoRef.current) {
          videoRef.current.currentTime = 0;
          handlePlayVideo(post.id); // Toggles to null, pausing it.
      }
  };

  const toggleMute = (e: React.MouseEvent) => {
    e.stopPropagation();
    setIsMuted(prev => !prev);
  };

  const handleSeek = (e: React.MouseEvent<HTMLDivElement>) => {
      e.stopPropagation();
      if (videoRef.current && progressRef.current) {
        const rect = progressRef.current.getBoundingClientRect();
        const pos = (e.clientX - rect.left) / rect.width;
        videoRef.current.currentTime = pos * videoDuration;
      }
  };

  const formatCount = (count: number): string => {
    if (count >= 1000000) {
      return (count / 1000000).toFixed(1).replace(/\.0$/, '') + 'M';
    }
    if (count >= 1000) {
      return (count / 1000).toFixed(1).replace(/\.0$/, '') + 'K';
    }
    return count.toString();
  };
  
  const formatDuration = (seconds: number) => {
    if (isNaN(seconds) || seconds < 0) return '0:00';
    const minutes = Math.floor(seconds / 60);
    const remainingSeconds = Math.floor(seconds % 60);
    return `${minutes}:${remainingSeconds.toString().padStart(2, '0')}`;
  };

  useEffect(() => {
    const handleClickOutside = (event: MouseEvent) => {
      if (menuRef.current && !menuRef.current.contains(event.target as Node)) {
        setMenuOpen(false);
      }
    };
    document.addEventListener('mousedown', handleClickOutside);
    return () => document.removeEventListener('mousedown', handleClickOutside);
  }, []);
  

  const timeAgo = (dateString: string) => {
    const date = new Date(dateString);
    const now = new Date();
    const seconds = Math.floor((now.getTime() - date.getTime()) / 1000);
    
    let interval = seconds / 31536000;
    if (interval > 1) return Math.floor(interval) + "y";
    interval = seconds / 2592000;
    if (interval > 1) return Math.floor(interval) + "m";
    interval = seconds / 86400;
    if (interval > 1) return Math.floor(interval) + "d";
    interval = seconds / 3600;
    if (interval > 1) return Math.floor(interval) + "h";
    interval = seconds / 60;
    if (interval > 1) return Math.floor(interval) + "m";
    return Math.floor(seconds) + "s";
  }
  
  const isSubscribed = subscribedToUserIds.includes(post.user.id);
  const isOwnPost = post.user.id === currentUser.id;

  const canViewContent = isOwnPost || 
    !post.isSubscriberOnly || 
    (post.isSubscriberOnly && isSubscribed);

  const handleCommunityClick = (e: React.MouseEvent) => {
    e.preventDefault();
    e.stopPropagation();
    const community = MOCK_COMMUNITIES.find(c => c.id === post.communityId);
    if (community) {
      viewCommunity(community);
    }
  };
  
  const renderFormattedContent = (text: string) => {
    const parts = text.split(/([#@][\w_]+)/g);
    return (
      <>
        {parts.map((part, index) => {
          if (part.startsWith('#')) {
            return (
              <button
                key={index}
                onClick={(e) => {
                  e.stopPropagation();
                  handleSearch(part);
                }}
                className="text-primary dark:text-violet-400 hover:underline focus:outline-none"
              >
                {part}
              </button>
            );
          }
          if (part.startsWith('@')) {
            const handle = part;
            const user = allUsers.find(u => u.handle.toLowerCase() === handle.toLowerCase());
            if (user) {
              return (
                <button
                  key={index}
                  onClick={(e) => {
                    e.stopPropagation();
                    handleViewProfile(user.id);
                  }}
                  className="text-primary dark:text-violet-400 hover:underline focus:outline-none"
                >
                  {part}
                </button>
              );
            }
          }
          return part;
        })}
      </>
    );
  };
  
  const handleNativeShare = async (e: React.MouseEvent) => {
    e.stopPropagation();
    setShareMenuOpen(false);
    const url = `https://cascade-app.dev/post/${post.id}`;
    const shareData = {
      title: `Post by ${post.user.name}`,
      text: post.content.substring(0, 120) + (post.content.length > 120 ? '...' : ''),
      url: url,
    };

    if (navigator.share) {
      try {
        await navigator.share(shareData);
      } catch (err) {
        if ((err as Error).name !== 'AbortError') {
            console.error("Error sharing post:", err);
        }
      }
    } else {
      try {
        await navigator.clipboard.writeText(url);
        addNotification('Link copied to clipboard!', 'info');
      } catch (err) {
        console.error('Failed to copy link:', err);
        addNotification('Failed to copy link.', 'info');
      }
    }
  };

  const handleCopyLink = (e: React.MouseEvent) => {
    e.stopPropagation();
    setShareMenuOpen(false);
    const url = `https://cascade-app.dev/post/${post.id}`;
    navigator.clipboard.writeText(url).then(() => {
        addNotification('Link copied to clipboard!', 'info');
    }).catch(err => {
        console.error('Failed to copy link:', err);
        addNotification('Failed to copy link.', 'info');
    });
  };

  const handleShareToCommunityClick = (e: React.MouseEvent) => {
    e.stopPropagation();
    setShareMenuOpen(false);
    handleShareToCommunity(post);
  };

  const renderLockedState = () => {
    return (
       <article className="p-4 border-b border-gray-200 dark:border-dark-border flex space-x-4">
        <div className="flex-shrink-0">
          <img src={post.user.avatarUrl} alt={`${post.user.name}'s avatar`} className="w-12 h-12 rounded-full" />
        </div>
        <div className="flex-1 min-w-0">
          <div className="flex justify-between items-center">
            <div className="flex items-center space-x-1 min-w-0">
              <span className="font-bold truncate">{post.user.name}</span>
               {post.user.verificationStatus === 'verified' && <VerifiedIcon className="w-5 h-5 text-primary flex-shrink-0" title="Verified Account" />}
              <span className="text-on-surface-secondary dark:text-dark-on-surface-secondary truncate">{post.user.handle}</span>
              <span className="text-on-surface-secondary dark:text-dark-on-surface-secondary">·</span>
              <span className="text-on-surface-secondary dark:text-dark-on-surface-secondary">{timeAgo(post.timestamp)}</span>
            </div>
            <MoreIcon className="w-5 h-5 text-on-surface-secondary dark:text-dark-on-surface-secondary" />
          </div>
          <div className="my-2 p-6 rounded-2xl border border-gray-200 dark:border-dark-border bg-surface dark:bg-dark-surface flex flex-col items-center text-center">
            <LockIcon className="w-8 h-8 text-on-surface-secondary dark:text-dark-on-surface-secondary mb-2" />
            <h3 className="font-bold text-lg">{t('post_subscriber_only')}</h3>
            <p className="text-on-surface-secondary dark:text-dark-on-surface-secondary text-sm mb-4">
                {t('post_subscriber_only_prompt', { name: post.user.name })}
            </p>
            <button 
              onClick={() => handleSubscribe(post.user.id)}
              className="bg-on-surface dark:bg-dark-on-surface text-background dark:text-dark-background font-bold px-6 py-2 rounded-full hover:bg-on-surface/90 dark:hover:bg-dark-on-surface/90 transition-colors duration-200"
            >
              {t('button_subscribe_for', { price: post.user.subscriptionPrice || 0 })}
            </button>
          </div>
        </div>
      </article>
    )
  }

  if (!canViewContent) {
    return renderLockedState();
  }
  
  const groupedGifts = (post.giftsReceived || []).reduce((acc, userGift) => {
    const giftInfo = MOCK_GIFTS.find(g => g.id === userGift.giftId);
    if(giftInfo) {
        if (!acc[giftInfo.id]) {
            acc[giftInfo.id] = { ...giftInfo, count: 0 };
        }
        acc[giftInfo.id].count++;
    }
    return acc;
  }, {} as Record<string, Gift & { count: number }>);

  const handleOpenDetails = (e: React.MouseEvent) => {
      e.stopPropagation();
      if (isDetailView) return;
      handleViewPost(post);
  };

  return (
    <>
      <article className="p-4 border-b border-gray-200 dark:border-dark-border hover:bg-gray-50 dark:hover:bg-dark-surface/30 transition-colors duration-200">
        <div className="flex space-x-4">
          <div className="flex-shrink-0">
            <img src={post.user.avatarUrl} alt={`${post.user.name}'s avatar`} className="w-12 h-12 rounded-full" />
          </div>
          <div className="flex-1 min-w-0">
            <div className="flex justify-between items-center">
              <div className="flex items-center space-x-1 min-w-0 text-on-surface-secondary dark:text-dark-on-surface-secondary">
                <button onClick={(e) => {e.stopPropagation(); handleViewProfile(post.user.id)}} className="font-bold truncate text-on-surface dark:text-dark-on-surface hover:underline">{post.user.name}</button>
                {post.user.verificationStatus === 'verified' && <VerifiedIcon className="w-5 h-5 text-primary flex-shrink-0" title="Verified Account" />}
                <span className="truncate">{post.user.handle}</span>
                {post.isSubscriberOnly && <StarIcon className="w-4 h-4 text-amber-500 ml-1" title="Subscriber-only post" />}
                <span>· {timeAgo(post.timestamp)}</span>
                {post.location && (
                  <>
                    <span>·</span>
                    <span className="flex items-center gap-1 truncate">
                      <LocationIcon className="w-4 h-4 flex-shrink-0" />
                      <span className="truncate">{post.location}</span>
                    </span>
                  </>
                )}
                {post.editedAt && <span className="italic text-xs">({t('post_edited')})</span>}
                {post.communityName && post.communityId && (
                  <>
                    <span>·</span>
                    <button onClick={handleCommunityClick} className="hover:underline">c/{post.communityName}</button>
                  </>
                )}
              </div>
               <div className="relative" ref={menuRef}>
                <button onClick={(e) => { e.stopPropagation(); setMenuOpen(!isMenuOpen); }} className="p-1 rounded-full hover:bg-primary/10 dark:hover:bg-primary/20">
                  <MoreIcon className="w-5 h-5 text-on-surface-secondary dark:text-dark-on-surface-secondary" />
                </button>
                {isMenuOpen && (
                  <div className="absolute top-full right-0 mt-1 w-56 bg-background dark:bg-dark-background rounded-lg shadow-lg border border-gray-200 dark:border-dark-border z-10 overflow-hidden">
                    <ul>
                      {isOwnPost && !isBoosted && handleOpenBoostModal && (
                        <li>
                          <button
                            onClick={(e) => { e.stopPropagation(); handleOpenBoostModal(post); setMenuOpen(false); }}
                            className="w-full text-left px-4 py-2 flex items-center gap-3 hover:bg-surface dark:hover:bg-dark-surface transition-colors duration-200 text-sm font-semibold"
                          >
                            <RocketIcon className="w-5 h-5" />
                            <span>Boost Post</span>
                          </button>
                        </li>
                      )}
                      {isOwnPost && (
                        <li>
                          <button
                            onClick={(e) => { e.stopPropagation(); handleOpenEditModal(post); setMenuOpen(false); }}
                            className="w-full text-left px-4 py-2 flex items-center gap-3 hover:bg-surface dark:hover:bg-dark-surface transition-colors duration-200 text-sm font-semibold"
                          >
                            <EditIcon className="w-5 h-5" />
                            <span>{t('post_menu_edit')}</span>
                          </button>
                        </li>
                      )}
                      <li>
                        <button
                          onClick={(e) => { e.stopPropagation(); handleToggleBookmark(post.id); setMenuOpen(false); }}
                          className="w-full text-left px-4 py-2 flex items-center gap-3 hover:bg-surface dark:hover:bg-dark-surface transition-colors duration-200 text-sm font-semibold"
                        >
                          {post.isBookmarked ? <BookmarkedIcon className="w-5 h-5 text-primary" /> : <BookmarkIcon className="w-5 h-5" />}
                          <span>{post.isBookmarked ? 'Remove Bookmark' : 'Bookmark Post'}</span>
                        </button>
                      </li>
                      {analyticsData && (
                        <li>
                          <button
                            onClick={(e) => { e.stopPropagation(); setAnalyticsModalOpen(true); setMenuOpen(false); }}
                            className="w-full text-left px-4 py-2 flex items-center gap-3 hover:bg-surface dark:hover:bg-dark-surface transition-colors duration-200 text-sm font-semibold"
                          >
                            <AnalyticsIcon className="w-5 h-5" />
                            <span>View Post Analytics</span>
                          </button>
                        </li>
                      )}
                      {!isOwnPost && post.user.monetizationSettings?.giftsEnabled && (
                        <li>
                          <button onClick={(e) => { e.stopPropagation(); openGiftModal(post); setMenuOpen(false); }} className="w-full text-left px-4 py-2 flex items-center gap-3 hover:bg-surface dark:hover:bg-dark-surface transition-colors duration-200 text-sm font-semibold">
                            <GiftIcon className="w-5 h-5"/>
                            <span>Send a Gift</span>
                          </button>
                        </li>
                      )}
                       {!isOwnPost && (
                        <>
                          <li><div className="h-px bg-gray-200 dark:bg-dark-border my-1" /></li>
                          <li><button onClick={(e) => { e.stopPropagation(); handleToggleBlock(post.user.id); setMenuOpen(false); }} className="w-full text-left px-4 py-2 hover:bg-surface dark:hover:bg-dark-surface transition-colors duration-200 text-sm font-semibold text-red-600">{blockedUserIds.includes(post.user.id) ? `Unblock ${post.user.handle}` : `Block ${post.user.handle}`}</button></li>
                        </>
                       )}
                    </ul>
                  </div>
                )}
              </div>
            </div>
            
            {isScheduled && (
                <div className="my-2 p-3 bg-blue-50 dark:bg-blue-900/20 border border-blue-200 dark:border-blue-800 rounded-lg text-sm text-blue-700 dark:text-blue-300 flex items-center gap-2">
                    <CalendarIcon className="w-5 h-5"/>
                    <span>Scheduled to post on {new Date(post.scheduledAt!).toLocaleString()}</span>
                </div>
            )}

            <div onClick={handleOpenDetails} className="cursor-pointer">
              {post.content && (
                  <p className="my-1 text-lg whitespace-pre-wrap break-words">{renderFormattedContent(post.content)}</p>
              )}
              
              {post.fileType === 'image' && post.imageUrl && (
                <div className="mt-3 rounded-2xl border border-gray-200 dark:border-dark-border overflow-hidden">
                  <img src={post.imageUrl} alt="Post content" className="w-full object-cover max-h-[600px] cursor-pointer" onClick={(e) => {e.stopPropagation(); setImageModalOpen(true)}} />
                </div>
              )}
              
              {post.fileType === 'video' && post.videoUrl && (
                <div className="mt-3 rounded-2xl border border-gray-200 dark:border-dark-border overflow-hidden relative group" onClick={(e) => {e.stopPropagation(); handlePlayVideo(post.id)}}>
                    <video ref={videoRef} src={post.videoUrl} poster={post.thumbnailUrl} className="block w-full object-cover max-h-[600px] bg-black" muted playsInline onTimeUpdate={handleTimeUpdate} onLoadedMetadata={handleLoadedMetadata} onEnded={handleVideoEnd} />
                    <div className={`absolute inset-0 bg-black/30 flex items-center justify-center transition-opacity duration-300 pointer-events-none ${showControls ? 'opacity-100' : 'opacity-0'}`}>
                        <button className="text-white p-2 bg-black/50 rounded-full pointer-events-auto">
                          {isPlaying ? <PauseIcon className="w-12 h-12" /> : <PlayIcon className="w-12 h-12" />}
                        </button>
                    </div>
                    {showControls && (
                      <div className="absolute bottom-2 left-2 right-2 p-2 pointer-events-none">
                        <div className="flex items-center gap-2">
                          <span className="text-white text-xs font-mono drop-shadow-md">{formatDuration(currentTime)}</span>
                          <div ref={progressRef} onClick={handleSeek} className="w-full h-1.5 bg-white/30 rounded-full cursor-pointer group pointer-events-auto">
                            <div className="h-full bg-white rounded-full relative" style={{ width: `${(currentTime / videoDuration) * 100}%` }}>
                              <div className="absolute right-0 top-1/2 -translate-y-1/2 w-3 h-3 bg-white rounded-full opacity-0 group-hover:opacity-100 transition-opacity"></div>
                            </div>
                          </div>
                          <span className="text-white text-xs font-mono drop-shadow-md">{formatDuration(videoDuration)}</span>
                          <button onClick={toggleMute} className="text-white pointer-events-auto p-1">
                              {isMuted ? <VolumeOffIcon className="w-5 h-5" /> : <VolumeUpIcon className="w-5 h-5" />}
                          </button>
                        </div>
                      </div>
                    )}
                </div>
              )}

              {post.quotedPost && (
                <div className="mt-2 border border-gray-200 dark:border-dark-border rounded-2xl p-3 cursor-pointer" onClick={(e) => { e.stopPropagation(); handleViewPost(post.quotedPost!)}}>
                   <div className="flex items-center gap-2"><img src={post.quotedPost.user.avatarUrl} className="w-5 h-5 rounded-full" /><span className="font-bold text-sm">{post.quotedPost.user.name}</span><span className="text-sm text-on-surface-secondary dark:text-dark-on-surface-secondary">{post.quotedPost.user.handle}</span></div>
                   <p className="text-sm mt-1 line-clamp-3">{post.quotedPost.content}</p>
                   {post.quotedPost.imageUrl && <img src={post.quotedPost.imageUrl} className="mt-2 rounded-lg" />}
                </div>
              )}
              
              {post.poll && <PollDisplay initialPoll={post.poll} />}
              {post.taggedProduct && <ProductCard product={post.taggedProduct} onAddToCart={handleAddToCart} />}
            </div>

            <div className="mt-4 flex justify-between items-center text-on-surface-secondary dark:text-dark-on-surface-secondary">
              <button onClick={(e) => { e.stopPropagation(); handleToggleLike(post.id); }} className="flex items-center space-x-2 hover:text-pink-500 group">
                {post.isLiked ? <LikedIcon className="w-5 h-5 text-pink-500"/> : <LikeIcon className="w-5 h-5 group-hover:text-pink-500"/>}
                <span className={`text-sm ${post.isLiked ? 'text-pink-500' : ''}`}>{formatCount(post.likes)}</span>
              </button>
              <button onClick={(e) => { e.stopPropagation(); handleViewPost(post) }} className="flex items-center space-x-2 hover:text-blue-500 group">
                <CommentIcon className="w-5 h-5 group-hover:text-blue-500"/>
                <span className="text-sm">{formatCount(post.comments)}</span>
              </button>
              <button onClick={(e) => { e.stopPropagation(); handleToggleEcho(post.id); }} className="flex items-center space-x-2 hover:text-green-500 group">
                <EchoIcon className={`w-5 h-5 group-hover:text-green-500 ${post.isEchoed ? 'text-green-500' : ''}`}/>
                <span className={`text-sm ${post.isEchoed ? 'text-green-500' : ''}`}>{formatCount(post.echos)}</span>
              </button>
              <button onClick={(e) => { e.stopPropagation(); setShareMenuOpen(true); }} className="flex items-center space-x-2 hover:text-primary group">
                <ShareIcon className="w-5 h-5 group-hover:text-primary"/>
              </button>
            </div>
            
            {Object.keys(groupedGifts).length > 0 && (
                <div className="mt-3 pt-3 border-t border-gray-200 dark:border-dark-border flex items-center gap-2 flex-wrap">
                    <p className="text-xs text-on-surface-secondary dark:text-dark-on-surface-secondary">Gifts:</p>
                    {Object.values(groupedGifts).map((gift: Gift & { count: number }) => (
                        <div key={gift.id} className="flex items-center gap-1 bg-surface dark:bg-dark-surface px-2 py-1 rounded-full text-xs">
                            <span className="text-base">{gift.icon}</span>
                            <span className="font-semibold">{gift.count > 1 ? gift.count : ''}</span>
                        </div>
                    ))}
                </div>
            )}
          </div>
        </div>
      </article>

      {isImageModalOpen && modalRoot && createPortal(
        <div className="fixed inset-0 bg-black/80 z-50 flex items-center justify-center" onClick={() => setImageModalOpen(false)}>
            <img src={post.imageUrl} className="max-w-full max-h-full object-contain" />
        </div>,
        modalRoot
      )}
      
      {modalRoot && createPortal(
        <ShareModal 
            isOpen={isShareMenuOpen}
            onClose={() => setShareMenuOpen(false)}
            post={post}
            handleEcho={handleToggleEcho}
            handleQuotePost={handleOpenQuoteModal}
            handleNativeShare={handleNativeShare}
            handleCopyLink={handleCopyLink}
            handleShareToCommunity={(e) => { e.stopPropagation(); handleShareToCommunity(post); setShareMenuOpen(false); }}
        />,
        modalRoot
      )}
      
      {modalRoot && createPortal(
        <PostAnalyticsModal 
            isOpen={isAnalyticsModalOpen}
            onClose={() => setAnalyticsModalOpen(false)}
            post={post}
        />,
        modalRoot
      )}
    </>
  );
};
